Slide 11: Understanding Authentication

LGA_CREDENTIALS = {
    "admin": "Admin",
    "ingawa": "Ingawa",
    "kankara": "Kankara"
}

def check_login(username):
    username_lower = username.lower()
    if username_lower == "admin":
        return 'admin', None
    if username_lower in LGA_CREDENTIALS:
        return 'lga', LGA_CREDENTIALS[username_lower]
    return None, None

Slide 12: Login Page Code

def login_page():
    st.title("🔐 Login")
    with st.form("login"):
        username = st.text_input("Username")
        submit = st.form_submit_button("Login")
        if submit:
            access, lga = check_login(username)
            if access:
                st.session_state['logged_in'] = True
                st.session_state['access_level'] = access
                st.session_state['lga_filter'] = lga
                st.rerun()

def clean_text_columns(df):
    cols = ['Q2. Local Government Area', 'Q3.Ward', 'Q4. Community Name']
    for col in cols:
        if col in df.columns:
            df[col] = df[col].str.replace('�', '').str.strip()

Slide 17: QC Check 1 - Age

def check_age_inconsistencies(df):
    issues = []
    age_problems = df[df['Years Living'] > df['HH Head Age']]
    for idx, row in age_problems.iterrows():
        issues.append({
            'LGA': row['LGA'],
            'Issue Type': 'Age Inconsistencies',
            'Description': f"Years Living > Age"
        })
    return issues

Slide 18: QC Check 2 - Duplicates

def check_duplicates(df):
    issues = []
    dups = df[df['unique_code'].duplicated(keep=False)]
    for idx, row in dups.iterrows():
        issues.append({
            'LGA': row['LGA'],
            'Issue Type': 'HH Duplicate',
            'Description': f"Duplicate: {row['unique_code']}"
        })
    return issues

Slide 19: QC Check 3 - Child Age

def check_child_age(child_df, main_df):
    issues = []
    problems = child_df[(child_df['Q94'] == 'Yes') & (child_df['Age'] > 59)]
    for idx, row in problems.iterrows():
        parent = main_df[main_df['_uuid'] == row['_submission__uuid']].iloc[0]
        issues.append({
            'LGA': parent['LGA'],
            'Issue Type': 'Child Age >59',
            'Description': f"Age: {row['Age']} months"
        })
    return issues

Slide 20: Complete QC Function

def perform_qc_checks(sheets_dict):
    all_issues = []
    main_df = sheets_dict['main']
    child_df = sheets_dict['child_infoo']
    
    all_issues.extend(check_age_inconsistencies(main_df))
    all_issues.extend(check_duplicates(main_df))
    all_issues.extend(check_child_age(child_df, main_df))
    
    return pd.DataFrame(all_issues)

Slide 27: Coverage Calculation

def calculate_coverage(df):
    coverage = []
    for code in df['Community Code'].unique():
        planned = COMMUNITY_DF[COMMUNITY_DF['Code'] == code]['Planned_HH'].values[0]
        reached = len(df[df['Community Code'] == code])
        pct = (reached / planned * 100) if planned > 0 else 0
        coverage.append({
            'Code': code, 'Planned': planned, 'Reached': reached,
            'Coverage %': pct, 'Status': '✅' if reached >= planned else '⚠️'
        })
    return pd.DataFrame(coverage)
